302  def compParDiff(self, name, chem, mass_frac_water, mass_frac_water_sat,
303  V_mortar, V_brick, V_all, T, eta) :
304  """ Compute the partition coefficient with respect to water
305  and the diffusion coefficient
306  """
307  rou_lipid = 0.9
308  rou_keratin = 1.37
309  rou_water = 1
310 
311  # 2. calulcate volume fraction of water in CC based on
312  # mass fraction of water in SC
313  theta_b = self.compVolFracWater_cc( mass_frac_water, rou_lipid, rou_keratin, rou_water,
314  V_mortar, V_brick, V_all)
315  # do the same for saturated water
316  phi_b = self.compVolFracWater_cc( mass_frac_water_sat, rou_lipid, rou_keratin, rou_water,
317  V_mortar, V_brick, V_all)
318 
319 
320  # 3. calculate diffusivity and partition coefficient
321 
322  K = 1.3806488 * 1E-23 # Boltzmann constant, Kg m^2 s^{-2}
323  r_f = 3.5e-9 # keratin microfibril radius, 3.5 nm
324  Dw = K*T/6/np.pi/eta/chem.r_s # diffusivity in water, Stoke-Eistein equation
325 
326  if name == 'LP' :
327  # m_Kw = pow(m_K_ow, 0.7) # this is the old model
328  Kw = rou_lipid / rou_water * (chem.K_ow ** 0.69)
329 
330  if chem.mw <= 380.0 :
331  r_s_inA = chem.r_s*1e10 # unit in Angstrom
332  D = 2 * 1E-9 * np.exp(-0.46*r_s_inA*r_s_inA)
333  else :
334  D = 3 * 1E-13
335  elif name == 'CC' : # corneocyte
336  # the following not used:
337  # if (m_K_ow>10)
338  # K_kw = 5.6 * pow(m_K_ow, 0.27);
339  # else
340  # K_kw = 0.5* ( 1 + pow(m_K_ow, 0.7) );
341  K_kw = rou_keratin / rou_water * 4.2 * (chem.K_ow**0.31)
342  Kw = (1-phi_b) * K_kw + theta_b
343 
344  # empirically fitted parameters
345  alpha = 9.47
346  beta = 9.32 * 1E-8
347  lambdaa = 1.09
348  gamma = -1.17
349 
350  r_s_inA = chem.r_s*1e10 # unit in A
351  r_f_inA = r_f*1e10 # unit in A
352 
353  phi_f = 1 - theta_b
354  k = beta*r_f_inA*r_f_inA* (phi_f**gamma)
355  S = (r_s_inA+r_f_inA)/r_f_inA
356  S = phi_f * S*S
357 
358  D = np.exp( -alpha*(S**lambdaa) ) / ( 1 + r_s_inA/np.sqrt(k) + r_s_inA*r_s_inA/3/k )
359  D *= Dw
360 
361  #comp.Comp.set_Kw(self, Kw)
362  #comp.Comp.set_D(self, Dw)
363 
364  return (Kw, D)

366  def compVolFracWater_cc(self, mass_frac_water, rou_lipid, rou_keratin, rou_water,
367  V_mortar_geometry, V_brick_geometry, V_all_geometry):
368  """ Compute the volume fraction of water in corneocyte
369  based on the water content (mass fraction of water) of the stratum corneum
370  """
371  f_l = 0.125 # dry mass fraction of SC lipid and keratin
372  f_k = 1 - f_l
373 
374  # mass fraction of lipid and keratin
375  mass_lipid = (1 - mass_frac_water) * f_l
376  mass_keratin = (1 - mass_frac_water) * f_k
377 
378  V_all = mass_lipid/rou_lipid + mass_keratin/rou_keratin + mass_frac_water/rou_water
379  V_lipid = mass_lipid/rou_lipid / V_all
380  V_keratin = mass_keratin/rou_keratin / V_all
381 
382  V_water_mortar = V_mortar_geometry/V_all_geometry - V_lipid
383  V_water_brick = V_brick_geometry/V_all_geometry - V_keratin
384 
385  vol_frac_water_cc = V_water_brick / V_brick_geometry * V_all_geometry
386
