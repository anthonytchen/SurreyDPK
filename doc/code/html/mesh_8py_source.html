94  """
95  if self.coord_sys == 'Cartesian' :
96  if direction==0 or direction==3 :
97  area = self.dy * self.dz
98  elif direction==1 or direction==2 :
99  area = self.dx * self.dz
100  else :
101  raise ValueError('direction must be one of: 0, 1, 2, 3')
102 
103  elif self.coord_sys == 'Cylindrical' :
104  r1 = self.y_coord
105  r2 = self.y_coord + self.dy;
106  pi_alpha_360 = np.pi * self.m_dz / 360
107  if direction==0 or direction==3 :

113  else :

116  else :
