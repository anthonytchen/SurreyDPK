196  def compTotalArea(self, direction):
197  """Compute self's total area to a certain direction
198  direction: [0] = up; [1] = left; [2] = right; [3] = down
199  """
200  if self.coord_sys == 'Catersian' :
201  if direction==0 or direction==3 :
202  area = self.y_length * self.dz_dtheta
203  elif direction==1 or direction==2 :
204  area = self.x_length * self.dz_dtheta
205  else:
206  raise ValueError('direction must be one of: 0, 1, 2, 3')
207 
208  elif self.coord_sys == 'Cylindrical' :
209  pi_alpha_360 = np.pi * self.dz_dtheta / 360
210  if direction==0 or direction==3 :
211  area = pi_alpha_360 * self.y_length * self.y_length
212  elif direction==1 :
213  area = 0
214  elif direction==2 :
215  area = self.x_length * pi_alpha_360 * 2 * self.y_length
216  else:
217  raise ValueError('direction must be one of: 0, 1, 2, 3')
218  else :
219  raise ValueError('Coordinate system not implemented')
220 
221  return area

271  def compMassIrregMeshRight(self, meshThis, conc_this):
272  """Compute mass transfer between meshThis and neighbouring meshes to the right
273  whose meshing does not match exactly to meshThis, e.g.
274  meshThis may interface with multiple meshess with self.meshBdyRight
275  """
276  bDone = False
277  massIntoThis = 0
278 
279  z_length = meshThis.compInterArea(2) / meshThis.get_dx() # interfacial z_length
280 
281  for i in range(self.n_meshBdyRight) :
282  x1_this = meshThis.get_x_coord()
283  x2_this = x1_this + meshThis.get_dx()
284 
285  currentX = self.meshBdyRight[i].get_x_coord() # x coordinates of the neighbouring mesh
286  nextX = currentX + self.meshBdyRight[i].get_dx()
287 
288  # to compare whether two real numbers are the same
289  if meshThis.get_dx()<self.meshBdyRight[i].get_dx():
290  thd = meshThis.get_dx()
291  else :
292  thd = self.meshBdyRight[i].get_dx()
293  thd *= 1e-3
294 
295 
296  if x1_this > currentX-thd and x1_this < nextX-thd :
297 
298  if x2_this < nextX+thd : # meshThis is contained between currentX & nextX
299  x_length = x2_this - x1_this
300  bDone = True
301  else : # meshThis extends beyond nextX
302  x_length = nextX - x1_this
303 
304  meshOther = self.meshBdyRight[i]
305  conc_other = meshOther.getConc()
306 
307  flux = meshThis.compFlux_diffu(meshOther, conc_this, conc_other,
308  meshThis.get_dy()/2, meshOther.get_dy()/2)
309 
310  meshThis.set_dx(x_length) # this is needed to calculate the correct interfacial area, especially for cylindrical coordinate
311  area = meshThis.compInterArea(2) # interfacial area to the right
312  mass = area * flux
313 
314  massIntoThis += mass
315  self.massOut_right[i] += -mass; # flux into meshThis is negative flux into the neighbour
316 
317  if bDone:
318  break
319  else:
320  meshThis.set_x_coord(nextX)
321  meshThis.set_dx(x2_this - x1_this - x_length)
322 
323  return massIntoThis

395  mass_transfer_rate = 0

397 
398  meshThis = self.meshes[idx_this]
399  conc_this = y[idx_this]
400  volume_this = meshThis.compVolume()
401 
402  # Setup the neighbouring grids and calculate the mass transfer rate.
403  # in the following order: up, left, (self), right, down.

412  raise ValueError('Invalid boundary condition')
413  else : # not topmost layer

415  meshUp = self.meshes[idx_other]
416  conc_other = y[idx_other]
417  flux = meshThis.compFlux_diffu(meshUp, conc_this, conc_other,
418  meshThis.get_dx()/2, meshUp.get_dx()/2)
419  area = meshThis.compInterArea(0) # interfacial area
420  mass_transfer_rate += area * flux
